--/ declaraciones /--
VAR
    [Integer, Float, Integer, String] : [a, b, c, x],
    [Integer, Float, Integer, String] : [s, r, t, y]
ENDVAR


--/ asignacion simple y expresion /--

c := 4;

t := 12;

x := "hola";

s := 1;

a := (((8 + c) + t) / 3) + 10;


--/ seleccion /--

IF (a<c) 
  c:=5;
ENDIF


--/ iteracion /--

REPEAT
    a := a+1;
    REPEAT
        c := c+1;
    UNTIL (NOT (c<t));
UNTIL (c>s OR t<a);


--/ print, read /--

PRINT a
READ b
