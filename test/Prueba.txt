--/ declaraciones /--
VAR
    [Integer, Float, Integer, String] : [a, b, c, x],
    [Integer, Integer, Integer, String] : [s, r, t, y]
ENDVAR


--/ asignacion simple y expresion /--

b := 2.5;

c := 4;

t := 12;

x := "hola";

a := (((8 + c)  + t) / 3) + 10;

--/ asignacion multiple /--

[r,s] := [100, 200]

--/ inlist /--

IF(INLIST(a,[2;4]))
    a:=4;
ENDIF

--/ iteracion /--

REPEAT
    a := a+1;
    REPEAT
        c := c+1;
    UNTIL (NOT (c<t));
UNTIL (c>s OR t<a);


--/ print, read /--

PRINT a
READ b
